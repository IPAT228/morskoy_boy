🧩 Конструктор пользовательской стратегии Противник

Настройте функцию выбора выстрела. Стратегия получает матрицу поля, список уже сделанных выстрелов и возвращает координаты следующей клетки для атаки.

Логика функции: противник перебирает все свободные клетки поля и для каждой рассчитывает score — числовую оценку выгодности выстрела. На score влияют выбранная чётность, близость клетки к центру, соседство с уже найденным попаданием и уровень случайности. После расчёта стратегия выбирает клетку с максимальным score. Перед сохранением тесты проверяют, что функция возвращает допустимые координаты, не повторяет выстрелы и корректно реагирует на попадание.

function scoreCustomCell(r, c, grid, shotSet, config) { let score = 0; const baseMode = 'weighted'; const parity = 'any'; const hitPriority = 8; const centerWeight = 2; const randomness = 3; if (parity === 'even' && (r + c) % 2 !== 0) score -= 8; if (parity === 'odd' && (r + c) % 2 !== 1) score -= 8; if (baseMode === 'checker' && (r + c) % 2 === 0) score += 4; const centerDistance = Math.abs(r - 4.5) + Math.abs(c - 4.5); score += (9 - centerDistance) * centerWeight; const nearHit = adjacentHitTargets(grid, shotSet).some(([nr, nc]) => nr === r && nc === c); if (nearHit) score += hitPriority * 5; score += Math.random() * randomness; return score; }
